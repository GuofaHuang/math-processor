	remove exportTable:
	exportTable   = function ( v_ColCount, b_BoolData) 
	{					
		if (count(b_BoolData) % v_ColCount == 0)
		{																	
			__start = 0;:			
			__body = "<table border=1 width=70% align=center>";:	
			__numRows = count(b_BoolData)/v_ColCount;:
			repeat (__numRows)
			{				
				__body = concat(__body, "<tr>");:				
				__rowOffset = 0;:					
				repeat (v_ColCount)
				{	
					__temp = part (b_BoolData, __start + __rowOffset * __numRows, 1);:					
					__temp = getstring(__temp);:
					__temp = concat ("<td align=center>", __temp, "</td>");:					
					__rowOffset = __rowOffset + 1;:				   
					__body = concat(__body, __temp);:
				}
				__body = concat (__body, "</tr>");:				
				__start = __start + 1;:																
			}			
			__body = concat(__body, "</table>");:				
			savefile(concat("<html><body>", __body, "</body></html>"));:
			remove __start __body __numRows __rowOffset __temp:
			echo ("Done. See the saved .html file."):
		}
		else 
		{
			echo("Boolean data count must be a multiple of the number of columns (your first parameter)."):
		}
	}
